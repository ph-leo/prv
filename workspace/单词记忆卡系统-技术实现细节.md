## 一、OCR识别模块实现

### 1.1 技术选型对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **PaddleOCR** | 中文识别准确率高，开源免费，支持表格识别 | 模型较大，推理速度一般 | 中文词汇表识别 ✅推荐 |
| **Tesseract** | 轻量级，多语言支持 | 中文识别率低 | 英文单词识别 |
| **百度AI OCR** | 准确率高，API简单易用 | 需要联网，有调用限制 | 云端部署 |
| **腾讯OCR** | 表格识别强，支持手写体 | 收费，有调用限制 | 企业级应用 |

# ocr_service.py
from paddleocr import PaddleOCR
import cv2
import numpy as np
import re
from typing import List, Dict

class WordOCRService:
    def __init__(self):
        # 初始化PaddleOCR
        self.ocr = PaddleOCR(
            use_angle_cls=True,  # 方向分类
            lang='ch',           # 中文模型
            use_gpu=False,       # CPU运行
            show_log=False
        )
    
    def preprocess_image(self, image_path: str) -> np.ndarray:
        """图片预处理"""
        # 读取图片
        img = cv2.imread(image_path)
        
        # 1. 旋转校正（如果图片倾斜）
        # 2. 去噪
        img = cv2.fastNlMeansDenoisingColored(img, None, 10, 10, 7, 21)
        
        # 3. 二值化（提高文字对比度）
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        _, binary = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
        
        # 4. 锐化
        kernel = np.array([[-1,-1,-1], [-1,9,-1], [-1,-1,-1]])
        sharpened = cv2.filter2D(binary, -1, kernel)
        
        return sharpened
    
    def detect_table_structure(self, image: np.ndarray) -> List[Dict]:
        """检测表格结构"""
        # 使用OpenCV检测表格线
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY) if len(image.shape) == 3 else image
        
        # 检测水平线和垂直线
        horizontal_kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (40, 1))
        vertical_kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (1, 40))
        
        horizontal_lines = cv2.morphologyEx(gray, cv2.MORPH_OPEN, horizontal_kernel)
        vertical_lines = cv2.morphologyEx(gray, cv2.MORPH_OPEN, vertical_kernel)
        
        # 合并线条
        table_structure = cv2.addWeighted(horizontal_lines, 0.5, vertical_lines, 0.5, 0.0)
        
        # 检测单元格
        contours, _ = cv2.findContours(table_structure, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
        
        cells = []
        for contour in contours:
            x, y, w, h = cv2.boundingRect(contour)
            if w > 50 and h > 20:  # 过滤小噪点
                cells.append({'x': x, 'y': y, 'w': w, 'h': h})
        
        return cells
    
    def extract_words(self, image_path: str) -> List[Dict]:
        """提取单词信息"""
        # 预处理
        processed_img = self.preprocess_image(image_path)
        
        # OCR识别
        result = self.ocr.ocr(image_path, cls=True)
        
        words = []
        for line in result[0]:
            if line:
                bbox = line[0]  # 边界框
                text = line[1][0]  # 识别的文字
                confidence = line[1][1]  # 置信度
                
                # 解析单词信息
                word_info = self.parse_word_text(text)
                if word_info:
                    word_info['confidence'] = confidence
                    word_info['bbox'] = bbox
                    words.append(word_info)
        
        return words
