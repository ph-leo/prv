```python
def get_prompt(body_part):
    body_lower = body_part.lower()
    if any(kw in body_lower for kw in ['胸', 'chest', '肺', '心']):
        return CHEST_PROMPT_V72, 'chest'
    # ...
```

**修改后**：
```python
def get_prompt(body_part, use_v74=True):
    if use_v74:
        # 检查骨折关键词
        fracture_keywords = ['骨折', '断裂', '骨皮质']
        if any(kw in body_lower for kw in fracture_keywords):
            return FRACTURE_PROMPT_V74, 'fracture'
        
        # 检查肺部关键词
        pulmonary_keywords = ['肺', '胸', '支气管', '肺炎']
        if any(kw in body_lower for kw in pulmonary_keywords):
            return PULMONARY_PROMPT_V74, 'pulmonary'
    
    # Fallback to V7.2 prompts
    # ...
```
